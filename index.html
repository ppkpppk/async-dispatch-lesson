Make callback optional

  function foo (value, callback) {
    setTimeout(function (arg, cb) {
      var returnValue = 'foo said:' + arg;
      kendoConsole.log(returnValue);
      if (cb) {
        cb(returnValue);
      }
    }, 1000, value, callback);
  };

  function bar (value) {
    kendoConsole.log('bar said:' + value);
  };

  foo('hello', bar);

Callback

Make sure the callback is a function

  function foo (value, callback) {
    setTimeout(function (arg, cb) {
      var returnValue = 'foo said:' + arg;
      kendoConsole.log(returnValue);
      if (cb && typeof(cb) === 'function') {
        cb(returnValue);
      }
    }, 1000, value, callback);
  };

  function bar (value) {
    kendoConsole.log('bar said:' + value);
  };

  foo('hello', bar);

Callback

Multi Callback with $.Callbacks

  function foo (value, callbacks) {
    setTimeout(function (arg, cb) {
      var returnValue = 'foo said:' + arg;
      kendoConsole.log(returnValue);
      cb.fire(returnValue);
    }, 1000, value, callbacks);
  };

  function bar (value) {
    kendoConsole.log('bar said:' + value);
  };

  function baz (value) {
    kendoConsole.log('baz said:' + value);
  };

  var callbacks = $.Callbacks();
  callbacks.add(bar);
  callbacks.add(baz);
  foo('hello', callbacks);

Callback

Pub/Sub

  var topics = {};

  function Topic (id) {
    var callbacks, topic = id && topics[id];
    if (!topic) {
      callbacks = $.Callbacks();
      topic = {
        publish: callbacks.fire,
        subscribe: callbacks.add,
        unsubscribe: callbacks.remove
      };
      if (id) {
        topics[id] = topic;
      }
    }
    return topic;
  };

  Topic('sayHello').subscribe(function (value) {kendoConsole.log('foo said:' + value);});
  Topic('sayHello').subscribe(function (value) {kendoConsole.log('bar said:' + value);});
  Topic('sayHello').publish('Hello world');

Promise

An async dispatch chain

  function foo (arg) {
    var value = arg;
    kendoConsole.log(value);
    setTimeout(function (arg) {
      var value = arg + ' bar';
      kendoConsole.log(value);
      setTimeout(function (arg) {
        kendoConsole.log(arg + ' baz');
      }, 1000, value);
    }, 1000, value);
  }
  $('button').on('click', function () {
    foo('foo');
  });

Promise

Return a deferred

  function foo (arg) {
    var deferred = $.Deferred(), value = arg;
    kendoConsole.log(value);
    setTimeout(function (arg) {
      deferred.resolve(arg);
    }, 1000, value);
    return deferred;
  }
  function bar (arg) {
    var deferred = $.Deferred(), value = arg + ' bar';
    kendoConsole.log(value);
    setTimeout(function (arg) {
      deferred.resolve(arg);
    }, 1000, value);
    return deferred;
  }
  $('button').on('click', function () {
    foo('foo').then(bar).done(function () {
        kendoConsole.log('finish!');
    });
  });

Promise

Waiting for input

  function foo () {
    var deferred = $.Deferred();
    kendoConsole.log('foo');

  function baz () {
    kendoConsole.log('baz');

  var step;
  $('#run').on('click', function () {
    step = foo();
    step.then(bar).done(bar);
  });
  $('#continue').on('click', function () {
    if (step) {
      step.resolve();
    }
  });
